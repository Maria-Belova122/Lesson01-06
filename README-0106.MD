my_dict = {'Москва' : [495, 499], 'Псков' : 8112, 'Самара' : 846} # Справочник телефонных кодов городов
print(my_dict)
print(my_dict.get('Псков')) # Вывести из справочника код Пскова
print(my_dict.get('Уфа', 'Такого города нет'))  # Вывести из справочника код Уфы
my_dict['Уфа'] = 347        # Новое значение справочника, Уфа добавляется в справочник
print(my_dict)
my_dict.update({'Орел' : 4862, 'Пермь' : 342})  # Добавление в справочник Орел и Пермь
print(my_dict)
del my_dict['Псков']        # Удалить из справочника Псков
print(my_dict)
a = my_dict.pop('Самара')   # Вычленить из справочника Самары, присвоение кода Самары переменной 'a'
print(a)        # Вывести значение кода Самары
print(my_dict)
